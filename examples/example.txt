## Example 7: Integration with Python

```python
# Basic Python integration example
import subprocess

def process_with_isolation(message):
    """Process message using module isolation system"""
    
    # Use shell script for module judgment
    module = subprocess.run(
        ["./scripts/judge_module.sh", message],
        capture_output=True,
        text=True
    ).stdout.strip()
    
    print(f"Module: {module}")
    
    # Use workflow enforcer
    result = subprocess.run(
        ["./scripts/workflow_enforcer.sh", message],
        capture_output=True,
        text=True
    )
    
    print(result.stdout)
    
    return module

# Test
process_with_isolation("How to write an abstract?")
```

## Example 8: Batch Processing

```bash
#!/bin/bash
# batch_process.sh

messages=(
    "文献综述的结构"
    "工程进度报告模板"
    "黄金市场分析"
    "Docker网络配置"
    "旅行计划建议"
)

for msg in "${messages[@]}"; do
    echo "========================================="
    echo "Processing: $msg"
    echo "-----------------------------------------"
    
    ./scripts/workflow_enforcer.sh "$msg" > /dev/null 2>&1
    
    module=$(cat .current_module 2>/dev/null || echo "unknown")
    echo "Processed in module: $module"
    echo ""
    
    sleep 1
done

echo "========================================="
echo "Batch processing complete"
echo "Module switch history:"
cat .module_switch_history 2>/dev/null || echo "No switch history"
```
